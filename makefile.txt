# Set the minimum version of CMake required
cmake_minimum_required(VERSION 3.10)

# Set your project's name and version
project(GameProject VERSION 1.0)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the executable for the main program
add_executable(Main main.cpp)

# Include directories for your header files if they are in separate directories
include_directories(.)

# Add each of your source files that needs to be compiled
# Replace 'FlexArray.cpp' with your actual source files
add_library(Enemy Enemy.cpp)
add_library(Game Game.cpp)
add_library(Object Object.cpp)
add_library(Objective Objective.cpp)
add_library(Player Player.cpp)
add_library(Room Room.cpp)

# Link the libraries with the executable
target_link_libraries(Main Enemy Game Object Objective Player Room)
